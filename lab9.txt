    mahang   VARCHAR2(10)  PRIMARY KEY,
    tenhang  VARCHAR2(100) NOT NULL,
    soluong  NUMBER(10)   DEFAULT 0 CHECK (soluong >= 0),

CREATE OR REPLACE TRIGGER trg_insert_hoadon
BEFORE INSERT ON hoadon
FOR EACH ROW
DECLARE
    v_soluong_ton  hang.soluong%TYPE;
    v_dem          NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_dem
    FROM   hang
    WHERE  mahang = :NEW.mahang;

    IF v_dem = 0 THEN
        RAISE_APPLICATION_ERROR(-20001,
            'Loi: Mahang [' || :NEW.mahang ||'] khong ton tai trong bang HANG.');
    END IF;
    SELECT soluong INTO v_soluong_ton
    FROM   hang
    WHERE  mahang = :NEW.mahang;
    IF :NEW.soluongban > v_soluong_ton THEN
        RAISE_APPLICATION_ERROR(-20002,
            'Loi: So luong ban (' || :NEW.soluongban ||
            ') vuot ton kho hien tai (' || v_soluong_ton ||') .');
    END IF;
    UPDATE hang
    SET    soluong = soluong - :NEW.soluongban
    WHERE  mahang  = :NEW.mahang;
END trg_insert_hoadon;
/

AFTER DELETE ON hoadon

    WHERE  mahang  = :OLD.mahang;

CREATE OR REPLACE TRIGGER trg_update_hoadon
BEFORE UPDATE OF soluongban ON hoadon
FOR EACH ROW
DECLARE
    v_soluong_ton  hang.soluong%TYPE;
    v_chenh_lech   NUMBER;
BEGIN
    v_chenh_lech := :NEW.soluongban - :OLD.soluongban;

    IF v_chenh_lech != 0 THEN        
        IF v_chenh_lech > 0 THEN
            SELECT soluong INTO v_soluong_ton
            FROM   hang
            WHERE  mahang = :NEW.mahang;

            IF v_chenh_lech > v_soluong_ton THEN
                RAISE_APPLICATION_ERROR(-20003,
                'Loi: Khong du ton kho. Can them: ' || v_chenh_lech ||
                    ', Con lai: ' || v_soluong_ton);
            END IF;
        END IF;
UPDATE hang
        SET    soluong = soluong - v_chenh_lech
        WHERE  mahang  = :NEW.mahang;
    END IF;
END trg_update_hoadon;
/

CREATE OR REPLACE PACKAGE pkg_nkbh_state AS
    g_row_count NUMBER := 0;
END pkg_nkbh_state;
/

CREATE OR REPLACE TRIGGER trg_nkbh_update_compound
FOR UPDATE ON nhatkybanhang
COMPOUND TRIGGER
    BEFORE STATEMENT IS
    BEGIN
        pkg_nkbh_state.g_row_count := 0;
    END BEFORE STATEMENT;
    BEFORE EACH ROW IS
    BEGIN
        pkg_nkbh_state.g_row_count := pkg_nkbh_state.g_row_count + 1;
        IF pkg_nkbh_state.g_row_count > 1 THEN
            RAISE_APPLICATION_ERROR(-20004,
                'Loi: Chi duoc UPDATE toi da 1 ban ghi mot lan.');
        END IF;
        UPDATE mathang
        SET    soluong = soluong - (:NEW.soluong - :OLD.soluong)
        WHERE  mahang  = :NEW.mahang;
    END BEFORE EACH ROW;

END trg_nkbh_update_compound;
/

-- f. Trigger UPDATE nâng cao – Kiểm tra nhiều điều kiện
CREATE OR REPLACE PACKAGE pkg_nkbh_upd2 AS
    g_row_count NUMBER := 0;
END pkg_nkbh_upd2;
/

CREATE OR REPLACE TRIGGER trg_nkbh_update_nangcao
FOR UPDATE ON nhatkybanhang
COMPOUND TRIGGER

    BEFORE STATEMENT IS
    BEGIN
        pkg_nkbh_upd2.g_row_count := 0;
    END BEFORE STATEMENT;

    BEFORE EACH ROW IS
        v_ton  mathang.soluong%TYPE;
    BEGIN
        pkg_nkbh_upd2.g_row_count := pkg_nkbh_upd2.g_row_count + 1;
        IF pkg_nkbh_upd2.g_row_count > 1 THEN
            RAISE_APPLICATION_ERROR(-20006,
                'Loi: Chi duoc UPDATE 1 ban ghi.');
        END IF;
        SELECT soluong INTO v_ton
        FROM   mathang WHERE mahang = :NEW.mahang;
        IF :NEW.soluong < v_ton THEN
            RAISE_APPLICATION_ERROR(-20007,
                'Loi: So luong cap nhat (' || :NEW.soluong ||
                ') nho hon ton kho hien tai (' || v_ton ||') .');
        END IF;
        IF :NEW.soluong = v_ton THEN
            RAISE_APPLICATION_ERROR(-20008,
                'Thong bao: So luong moi bang ton kho, khong can cap nhat.');
        END IF;
        UPDATE mathang
        SET    soluong = soluong - (:NEW.soluong - :OLD.soluong)
        WHERE  mahang  = :NEW.mahang;
    END BEFORE EACH ROW;

END trg_nkbh_update_nangcao;
/

-- g. Thủ tục xóa MATHANG (tác động 2 bảng)
CREATE OR REPLACE PROCEDURE proc_xoa_mathang (
    p_mahang IN mathang.mahang%TYPE
) AS
    v_dem NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_dem
    FROM   mathang
    WHERE  mahang = p_mahang;

    IF v_dem = 0 THEN
        DBMS_OUTPUT.PUT_LINE('Thong bao: Mahang [' || p_mahang ||
                             '] khong ton tai trong he thong.');
        RETURN;
    END IF;

    DELETE FROM nhatkybanhang WHERE mahang = p_mahang;
    DBMS_OUTPUT.PUT_LINE('Da xoa ' || SQL%ROWCOUNT ||
                         ' ban ghi trong NHATKYBANHANG.');

 
    DELETE FROM mathang WHERE mahang = p_mahang;
    DBMS_OUTPUT.PUT_LINE('Da xoa mahang [' || p_mahang ||
                         '] khoi bang MATHANG.');
    COMMIT;

EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        DBMS_OUTPUT.PUT_LINE('Loi: ' || SQLERRM);
END proc_xoa_mathang;
/

-- h. Hàm tính tổng tiền theo tên hàng
CREATE OR REPLACE FUNCTION fn_tongtien (
    p_tenhang IN mathang.tenhang%TYPE
) RETURN NUMBER AS
    v_tong NUMBER := 0;
BEGIN
    SELECT NVL(SUM(nk.soluong * nk.giaban), 0)
    INTO   v_tong
    FROM   nhatkybanhang nk
    JOIN   mathang       mh ON nk.mahang = mh.mahang
    WHERE  mh.tenhang = p_tenhang;

    RETURN v_tong;
END fn_tongtien;
/

-- a Trigger INSERT trên bảng DANGKY
CREATE OR REPLACE TRIGGER trg_insert_dangky
BEFORE INSERT ON dangky
FOR EACH ROW
DECLARE
    v_dem_lop   NUMBER;
    v_dem_sv    NUMBER;
    v_sotoida   lophoc.sisotoida%TYPE;
    v_sohientai lophoc.sisohientai%TYPE;
BEGIN
    SELECT COUNT(*) INTO v_dem_lop
    FROM   lophoc
    WHERE  malop = :NEW.malop;

    IF v_dem_lop = 0 THEN
        RAISE_APPLICATION_ERROR(-20001,
            'Loi: Malop [' || :NEW.malop ||'] khong ton tai trong bang LOPHOC.');
    END IF;
    SELECT COUNT(*) INTO v_dem_sv
    FROM   dangky
    WHERE  masv  = :NEW.masv
    AND    malop = :NEW.malop;

    IF v_dem_sv > 0 THEN
        RAISE_APPLICATION_ERROR(-20002,
            'Loi: Sinh vien [' || :NEW.masv ||
            '] da dang ky lop [' || :NEW.malop ||'] roi.');
    END IF;
    SELECT sisotoida, sisohientai
    INTO   v_sotoida, v_sohientai
    FROM   lophoc
    WHERE  malop = :NEW.malop;

    IF v_sohientai >= v_sotoida THEN
        RAISE_APPLICATION_ERROR(-20003,
            'Loi: Lop [' || :NEW.malop ||
            '] da day (' || v_sohientai ||'/' || v_sotoida ||') .');
    END IF;
    UPDATE lophoc
    SET    sisohientai = sisohientai + 1
    WHERE  malop = :NEW.malop;
END trg_insert_dangky;
/

-- Trigger UPDATE trên bảng DANGKY
CREATE OR REPLACE TRIGGER trg_update_dangky
BEFORE UPDATE OF malop ON dangky
FOR EACH ROW
DECLARE
    v_dem_lop   NUMBER;
    v_dem_sv    NUMBER;
    v_sotoida   lophoc.sisotoida%TYPE;
    v_sohientai lophoc.sisohientai%TYPE;
BEGIN
    IF :NEW.malop = :OLD.malop THEN
        RETURN;
    END IF;
    SELECT COUNT(*) INTO v_dem_lop
    FROM   lophoc
    WHERE  malop = :NEW.malop;

    IF v_dem_lop = 0 THEN
        RAISE_APPLICATION_ERROR(-20004,
            'Loi: Malop moi [' || :NEW.malop ||'] khong ton tai.');
    END IF;
    SELECT COUNT(*) INTO v_dem_sv
    FROM   dangky
    WHERE  masv  = :NEW.masv
    AND    malop = :NEW.malop;

    IF v_dem_sv > 0 THEN
        RAISE_APPLICATION_ERROR(-20005,
            'Loi: Sinh vien [' || :NEW.masv ||
            '] da ton tai trong lop moi [' || :NEW.malop ||'] .');
    END IF;
    SELECT sisotoida, sisohientai
    INTO   v_sotoida, v_sohientai
    FROM   lophoc
    WHERE  malop = :NEW.malop;

    IF v_sohientai >= v_sotoida THEN
        RAISE_APPLICATION_ERROR(-20006,
            'Loi: Lop moi [' || :NEW.malop ||
            '] da day (' || v_sohientai ||'/' || v_sotoida ||') .');
    END IF;
    UPDATE lophoc
    SET    sisohientai = sisohientai - 1
    WHERE  malop = :OLD.malop;

    UPDATE lophoc
    SET    sisohientai = sisohientai + 1
    WHERE  malop = :NEW.malop;
END trg_update_dangky;
/
